//crate that we want to use to read from stdin 
use std::io;

//entry point to our program 
fn main() {
    //two print statements to stdout 
    println!("Guess the number!");
    println!("Please input your guess.");

    //new string variable to store guess 
    let mut guess = String::new();

    //read from stdin and store in guess if valid otherwise fail 
    io::stdin().read_line(&mut guess)
        .expect("Failed to read line");

    //print that a sucessfull guess was made 
    println!("You guessed: {}", guess);
}
